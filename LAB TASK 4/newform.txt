  <script>
    const canvas = document.getElementById("myCanvas");
    const ctx = canvas.getContext("2d");

    // Gradient background
    const gradient = ctx.createLinearGradient(0, 0, 500, 300);
    gradient.addColorStop(0, "#fff3e0");
    gradient.addColorStop(1, "#ffe0b2");
    ctx.fillStyle = gradient;
    ctx.fillRect(0, 0, 500, 300);

    // Smiley face
    function drawSmiley() {
      ctx.beginPath();
      ctx.arc(250, 150, 80, 0, Math.PI * 2);
      ctx.fillStyle = "#FFEB3B";
      ctx.shadowBlur = 20;
      ctx.shadowColor = "#FFD740";
      ctx.fill();
      ctx.strokeStyle = "#FBC02D";
      ctx.lineWidth = 4;
      ctx.stroke();

      ctx.shadowBlur = 0;
      ctx.beginPath();
      ctx.arc(220, 130, 10, 0, Math.PI * 2);
      ctx.arc(280, 130, 10, 0, Math.PI * 2);
      ctx.fillStyle = "#333";
      ctx.fill();

      ctx.beginPath();
      ctx.arc(250, 160, 45, 0, Math.PI, false);
      ctx.lineWidth = 6;
      ctx.strokeStyle = "#333";
      ctx.stroke();

      ctx.font = "bold 26px Poppins";
      ctx.fillStyle = "#333";
      ctx.textAlign = "center";
      ctx.fillText("Thank You for Your Feedback!", 250, 280);
    }

    drawSmiley();
  </script>
